ARG YB_VERSION=2025.1.3.2-b1

# ── STAGE 1: BUILDER — AlmaLinux 8, mesma glibc do YugabyteDB ────────────
FROM almalinux:8 AS postgis-builder

# Instalar repositório PGDG oficial (RPMs pré-compilados, glibc 2.28)
RUN dnf install -y \
        https://download.postgresql.org/pub/repos/yum/reporpms/EL-8-x86_64/pgdg-redhat-repo-latest.noarch.rpm && \
    dnf install -y epel-release && \
    dnf config-manager --set-enabled powertools && \
    dnf -qy module disable postgresql && \
    # Instalar PostgreSQL 15 + PostGIS via RPM — sem compilar nada
    dnf install -y \
        postgresql15 \
        postgresql15-libs \
        postgis34_15 \
        postgis34_15-client && \
    dnf clean all

# Isolar apenas os artefatos que o YugabyteDB precisa
RUN mkdir -p /artifacts/lib /artifacts/extension && \
    # .so das funções geoespaciais
    cp /usr/pgsql-15/lib/postgis-*.so          /artifacts/lib/ && \
    cp /usr/pgsql-15/lib/postgis_topology-*.so /artifacts/lib/ 2>/dev/null || true && \
    cp /usr/pgsql-15/lib/rtpostgis-*.so        /artifacts/lib/ 2>/dev/null || true && \
    # arquivos de extensão (.control + .sql)
    cp /usr/pgsql-15/share/extension/postgis*  /artifacts/extension/ && \
    # libs de runtime que o .so precisa encontrar (GEOS, PROJ, GDAL)
    cp /usr/lib64/libgeos*.so*                 /artifacts/lib/ && \
    cp /usr/lib64/libproj*.so*                 /artifacts/lib/ && \
    cp /usr/lib64/libgdal*.so*                 /artifacts/lib/ && \
    cp /usr/lib64/libjson-c*.so*               /artifacts/lib/ && \
    cp /usr/lib64/libprotobuf-c*.so*           /artifacts/lib/ 2>/dev/null || true

# ── STAGE 2: RUNTIME — YugabyteDB puro ───────────────────────────────────
FROM yugabytedb/yugabyte:${YB_VERSION}

LABEL cascata.image.variant="full"
LABEL cascata.extensions.cat3="postgis,postgis_topology"

USER root

# Copiar artefatos do builder — mesma glibc, zero incompatibilidade
COPY --from=postgis-builder /artifacts/lib/       /home/yugabyte/postgres/lib/
COPY --from=postgis-builder /artifacts/extension/ /home/yugabyte/postgres/share/extension/

# pg_cron já está na imagem base — apenas configurar preload
ENV YB_TSERVER_FLAGS="ysql_pg_conf_csv=\"shared_preload_libraries='pg_cron,pg_stat_statements,pgaudit'\""

USER yugabyte
